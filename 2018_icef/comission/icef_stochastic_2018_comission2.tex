\section*{Stochastic Calculus}


\begin{enumerate}



\item Two stochastic processes are defined by the system of SDE:
\[
\begin{cases}
dX_t=(2+5t+X_t)dt+3dW_{t} \\
dY_t=4X_t dt+8Y_t dW_{t}
\end{cases}
\]
\begin{enumerate}
  \item Calculate $d(X_t^2 Y_t)$.
  \item Is $X_t^2 Y_t$ a martingale?
\end{enumerate}

\item Consider the following stochastic differential equation
\[
dX_t=-0.5e^{-2X_t}dt+e^{-X_t}dW_t,
\]
with deterministic initial value $X_0$.
\begin{enumerate}
\item Is $X_t$ a martingale?
\item Using the substitution $Y_t=f(X_t)$ solve this differential equation.
\end{enumerate}
Hint: try to find a function $f$ such that the term before $dt$ cancels out.



\end{enumerate}
